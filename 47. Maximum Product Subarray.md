```java
long maxProduct(int[] arr, int n) 
{
    if(n==1)
    {
       return arr[0];
    }

    long res = 0;
    long maxi=1,mini=1;

    for (int i=0;i<n;i++) 
    {
        if (arr[i] ==0) 
        {
            maxi = 1;
            mini = 1;
            continue;
        }

        long temp = maxi * arr[i];
        maxi = Math.max(arr[i],Math.max(maxi*arr[i], mini*arr[i]));
        mini = Math.min(arr[i],Math.min(temp, mini*arr[i]));
        res = Math.max(res,maxi);
    }

    return res;

}

```
