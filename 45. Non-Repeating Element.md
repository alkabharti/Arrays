```java
public int firstNonRepeating(int arr[], int n) 
{ 
    HashMap<Integer,Integer> map=new HashMap<Integer,Integer>();
    int i,ans=0;
    for(i=0;i<n;i++)
    {
        if(!map.containsKey(arr[i]))
            map.put(arr[i],1);
        else
            map.put(arr[i],map.get(arr[i])+1);
    }
    for(i=0;i<n;i++)
    {
        if(map.get(arr[i])==1)
            return arr[i];
    }
    return ans;
}  

```
